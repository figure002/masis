1 /**
  2  * Round a decimal number to a set number of decimals.
  3  *
  4  * @param {Number} num The number to be rounded
  5  * @param {Number} dec The number of decimals to round to
  6  * @return {Number} The rounded number
  7  */
  8 function roundNumber(num, dec) {
  9 	var result = Math.round(num*Math.pow(10,dec))/Math.pow(10,dec);
 10 	return result;
 11 }

 13 /**
 14  * Populate a HTML select element with option elements.
 15  *
 16  * @param {Object} selobj The HTML select element object
 17  * @param {String} url The URL which will return JSON formatted options
 18  * @param {Object} callback The callback function to be called after the select
 19  *      element is populated
 20  */
 21 function loadSelectList(selobj, url, callback)
 22 {
 23     selobj.empty();
 24     $.getJSON(url, {}, function(data) {
 25         // Build the select list.
 26         $.each(data, function(i,obj) {
 27             selobj.append($('<option></option>')
 28                 .html(obj.label)
 29                 .attr('label', obj.value));
 30         });
 31         // Run callback method when done.
 32         if (callback) {
 33             selobj.promise().done(callback);
 34         }
 35     });
 36 }
